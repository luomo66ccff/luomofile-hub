API

Developer Docs

All examples use scoped API keys. Keep keys secret.

Upload file

curl -H "Authorization: Bearer lfk_xxx" \
  -F "file=@example.png" \
  -F "purpose=image" \
  -F "generate_link=false" \
  {{ base_url }}/api/v1/files/upload

List files

curl -H "Authorization: Bearer lfk_xxx" \
  {{ base_url }}/api/v1/files

Get file detail

curl -H "Authorization: Bearer lfk_xxx" \
  {{ base_url }}/api/v1/files/lf_xxx

Generate direct link

curl -X POST -H "Authorization: Bearer lfk_xxx" \
  {{ base_url }}/api/v1/files/lf_xxx/link/generate

Revoke direct link

curl -X POST -H "Authorization: Bearer lfk_xxx" \
  {{ base_url }}/api/v1/files/lf_xxx/link/revoke

Delete file

curl -X DELETE -H "Authorization: Bearer lfk_xxx" \
  {{ base_url }}/api/v1/files/lf_xxx

Storage stats

curl -H "Authorization: Bearer lfk_xxx" \
  {{ base_url }}/api/v1/storage/stats
